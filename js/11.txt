 * - width: chi?u r?ng, là 1 s? b?t kì

 * Vi?t các method:

var rectangle = {
  width : 8,
  height : 8,
  getwidth : function() 
  {
    console.log(this.width);
  },
  getheight : function()
  {
    console.log(this.height);
