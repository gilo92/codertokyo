// Sử dụng ternary operator thay thế cho if else sao cho đoạn code sau không thay đổi logic
function doSomething(a) {
  var b;
  if (a > 0) {
    b = 1;
  } else if (a < 0) {
    b = 2;
  } else {
    b = 3;
  }
  console.log(b);
  return b;
  
}

function doSomethingTernary(a) {
  var b;
  a > 0 ? b =1 : a < 0 ? b=2 : b=3;
  console.log(b);
  return b;
}

doSomething(-9);
doSomethingTernary(-9);
doSomething(0);
doSomethingTernary(0);
